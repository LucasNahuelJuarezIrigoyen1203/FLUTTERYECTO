# 🚀 BACKEND FASTAPI

# Activar entorno virtual (si usás venv)
source venv/bin/activate  # Linux/macOS
venv\Scripts\activate     # Windows

# Instalar dependencias (si no están)
pip install fastapi uvicorn python-dotenv bcrypt pyodbc

# Ejecutar el servidor FastAPI
uvicorn main:app --host 0.0.0.0 --port 8000 --reload


# 🌐 FRONTEND FLUTTER WEB

# Navegar al proyecto Flutter
cd khroma_frontend  # o el nombre de tu carpeta

# Instalar dependencias
flutter pub get

# Ejecutar en Chrome (Flutter Web)
flutter run -d chrome

# (Opcional) Ejecutar en dispositivo físico Android
flutter devices  # para ver ID del dispositivo
flutter run -d <ID_DEL_DISPOSITIVO>
